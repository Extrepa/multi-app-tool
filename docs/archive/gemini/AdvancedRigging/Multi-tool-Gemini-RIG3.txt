```javascript
// Phase 7: Reactivity Injection
let targetIntensity = 0.03; // Base 'Breathe' intensity
let currentIntensity = 0.03;

// 1. Hover: Increase 'Breath' rate
el.addEventListener('mouseenter', () => {
  targetIntensity = 0.15; // Deep breathing (excited)
});

el.addEventListener('mouseleave', () => {
  targetIntensity = 0.03; // Calm breathing
});

// 2. Click: High-Intensity 'Shake' (The Startle)
el.addEventListener('mousedown', () => {
  startleStartTime = performance.now();
  isStartled = true;
});

// Inside the Animation Loop:
function animate() {
  const t = (performance.now() - start) / 1000;
  
  // Smoothly interpolate between intensities (Surgical Easing)
  currentIntensity += (targetIntensity - currentIntensity) * 0.1;
  
  // Calculate Startle Jitter
  let jitter = 0;
  if (isStartled) {
    const elapsed = (performance.now() - startleStartTime) / 1000;
    if (elapsed < 0.5) {
      jitter = Math.sin(t * 50) * 10 * (1 - elapsed / 0.5); // Decay shake
    } else {
      isStartled = false;
    }
  }

  // Apply combined transforms
  el.style.transform = `translate(${jitter}px, ${floatY}px) scale(${1 + Math.sin(t * 0.5) * currentIntensity})`;
}

```
