export const VibeOscillators = {
  float: (t: number, freq: number, amp: number, phase: number) => {
    return Math.sin(t * freq + phase) * amp;
  },
  breathe: (t: number, freq: number, amp: number) => {
    // Non-uniform scaling for organic feel
    const scaleX = 1 + Math.sin(t * freq) * (amp * 0.05);
    const scaleY = 1 + Math.sin(t * freq + Math.PI) * (amp * 0.02);
    return { scaleX, scaleY };
  }
};
