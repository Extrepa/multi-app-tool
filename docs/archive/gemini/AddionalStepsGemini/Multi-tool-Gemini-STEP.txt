### **3. The Command Center Dashboard (`App.tsx`)**

The primary interface provides a mission-control view, featuring progress tracking, level progression, and detailed phase task lists.

```tsx
import React, { useState, useEffect } from 'react';
import { 
  Terminal, LayoutDashboard, ListTodo, Target, ArrowRight, CheckCircle
} from 'lucide-react';
import { PROJECT_PHASES } from './constants';

export default function App() {
  const [completedSteps, setCompletedSteps] = useState<number[]>([]);
  const [activeTab, setActiveTab] = useState<'dashboard' | number>('dashboard');

  const toggleStep = (id: number) => {
    setCompletedSteps(prev => prev.includes(id) ? prev.filter(i => i !== id) : [...prev, id]);
  };

  const progress = Math.round((completedSteps.length / 250) * 100);

  return (
    <div className="min-h-screen bg-[#050508] text-[#e0e0e0] flex font-sans">
      <aside className="w-72 bg-[#13131f] border-r border-gray-800 flex flex-col h-screen sticky top-0">
        <div className="p-6 border-b border-gray-800 flex items-center gap-3 text-[#00ff9d]">
          <Terminal size={24} />
          <span className="font-mono font-bold tracking-wider">STUDIO OS v1.0</span>
        </div>
        <nav className="flex-1 p-4 space-y-1 overflow-y-auto">
          <button onClick={() => setActiveTab('dashboard')} className="w-full text-left px-4 py-3 rounded-lg text-sm flex gap-3 text-[#b026ff]">
            <LayoutDashboard size={18} /> Mission Control
          </button>
          <div className="pt-6 pb-2 px-4 text-[10px] text-gray-600 font-mono uppercase">Build Phases</div>
          {PROJECT_PHASES.map(p => (
            <button key={p.id} onClick={() => setActiveTab(p.id)} className="w-full text-left px-4 py-2 text-xs text-gray-500 hover:text-white">
              {p.title}
            </button>
          ))}
        </nav>
      </aside>

      <main className="flex-1 p-12 overflow-y-auto max-w-6xl mx-auto">
        {activeTab === 'dashboard' ? (
          <div className="space-y-8 animate-in fade-in duration-500">
            <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
              <div className="bg-[#13131f] border border-gray-800 rounded-2xl p-6">
                <h2 className="text-xs font-mono text-[#00ff9d] mb-2 uppercase tracking-widest">Build Progress</h2>
                <div className="text-4xl font-black mb-4">{progress}%</div>
                <div className="w-full bg-gray-800 h-2 rounded-full overflow-hidden">
                  <div className="h-full bg-gradient-to-r from-[#b026ff] to-[#00ff9d]" style={{ width: `${progress}%` }} />
                </div>
              </div>
            </div>
          </div>
        ) : (
          <div className="animate-in slide-in-from-right-4 duration-300">
             <h1 className="text-3xl font-black mb-6 uppercase">Phase {activeTab}</h1>
             <div className="space-y-3">
               {PROJECT_PHASES.find(p => p.id === activeTab)?.steps.map(step => (
                 <div 
                   key={step.id} 
                   onClick={() => toggleStep(step.id)}
                   className={`p-4 border rounded-xl flex items-center gap-4 cursor-pointer transition-all ${completedSteps.includes(step.id) ? 'bg-[#00ff9d]/5 border-[#00ff9d]/30 text-gray-500' : 'bg-[#13131f] border-gray-800'}`}
                 >
                   <CheckCircle className={completedSteps.includes(step.id) ? 'text-[#00ff9d]' : 'text-gray-700'} />
                   <span className={completedSteps.includes(step.id) ? 'line-through' : ''}>{step.text}</span>
                 </div>
               ))}
             </div>
          </div>
        )}
      </main>
    </div>
  );
}

```
